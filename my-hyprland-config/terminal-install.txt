mkdir -p ~/.config/hypr ~/.config/waybar ~/.config/wofi

# Hyprland Config
cat > ~/.config/hypr/hyprland.conf << 'EOF'
# Hyprland config - i3-style, English, black & white, with animations

# Monitor
monitor=,preferred,auto,1

# Mod key
$mod = ALT

# Terminal
$term = kitty

# App launcher
exec-once = dbus-update-activation-environment --systemd WAYLAND_DISPLAY XDG_CURRENT_DESKTOP
exec-once = swaybg -i /usr/share/backgrounds/default.jpg -m fill
exec-once = waybar

# Input
input {
  kb_layout = us
  follow_mouse = 1
  sensitivity = 0
  repeat_rate = 50
  repeat_delay = 300
}

# General
general {
  gaps_in = 6
  gaps_out = 12
  border_size = 2
  col.active_border = rgba(ffffffcc) rgba(000000cc) 45deg
  col.inactive_border = rgba(55555588)
  layout = dwindle
}

# Animations
animations {
  enabled = true
  bezier = myBezier, 0.3, 1, 0.7, 1
  animation = windows, 1, 7, myBezier
  animation = windowsOut, 1, 7, myBezier
  animation = border, 1, 10, default
  animation = fade, 1, 7, default
  animation = workspaces, 1, 5, default
}

# Decoration
decoration {
  rounding = 6

  blur {
    enabled = false
    size = 3
    passes = 1
    ignore_opacity = true
  }
}

# Bindings - Launch & Manage
bind = $mod, RETURN, exec, $term
bind = $mod, A, exec, wofi --show drun
bind = $mod, Q, killactive
bind = $mod, F, togglefloating
bind = $mod, SPACE, togglefloating

# Focus movement (i3-style)
bind = $mod, H, movefocus, l
bind = $mod, L, movefocus, r
bind = $mod, J, movefocus, d
bind = $mod, K, movefocus, u

# Move windows
bind = $mod SHIFT, H, movewindow, l
bind = $mod SHIFT, L, movewindow, r
bind = $mod SHIFT, J, movewindow, d
bind = $mod SHIFT, K, movewindow, u

# Workspaces
bind = $mod, 1, workspace, 1
bind = $mod, 2, workspace, 2
bind = $mod, 3, workspace, 3
bind = $mod, 4, workspace, 4
bind = $mod, 5, workspace, 5
bind = $mod, 6, workspace, 6
bind = $mod, 7, workspace, 7
bind = $mod, 8, workspace, 8
bind = $mod, 9, workspace, 9
bind = $mod, 0, workspace, 10

# Mouse actions
bindm = $mod, mouse:272, movewindow
bindm = $mod, mouse:273, resizewindow
EOF

# Waybar Config
cat > ~/.config/waybar/config << 'EOF'
{
  "layer": "top",
  "position": "top",
  "height": 30,
  "modules-left": ["workspaces"],
  "modules-center": ["clock"],
  "modules-right": ["pulseaudio", "network", "battery"],
  "workspaces": {
    "format": "{icon} {name}"
  },
  "clock": {
    "format": "{:%H:%M  %a, %d %b %Y}"
  },
  "battery": {
    "format": "{capacity}% {icon}",
    "format-icons": ["", "", "", "", ""]
  },
  "pulseaudio": {
    "format": "{volume}% {icon}",
    "format-icons": ["", "", ""]
  },
  "network": {
    "format-wifi": "{signalStrength}% ",
    "format-ethernet": "",
    "format-disconnected": "⚠️"
  }
}
EOF

# Waybar Style
cat > ~/.config/waybar/style.css << 'EOF'
* {
  font-family: "JetBrainsMono Nerd Font", "monospace";
  font-size: 13px;
  color: #ffffff;
  background-color: rgba(0, 0, 0, 0.7);
  border: none;
}

window#waybar {
  background-color: rgba(0, 0, 0, 0.85);
  border-bottom: 2px solid white;
}

#clock, #battery, #pulseaudio, #network, #workspaces {
  padding: 0 12px;
}
EOF

# Wofi Style (Black & White)
cat > ~/.config/wofi/style.css << 'EOF'
window {
  background-color: #1e1e1e;
  color: #ffffff;
  font-family: JetBrainsMono Nerd Font;
  font-size: 14px;
  border: 2px solid white;
}
EOF

echo "✅ Done! Hyprland config installed with animations and English UI."
echo "🔁 Restart Hyprland or relogin to apply changes."
